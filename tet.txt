local info = redis.call("INFO", "stats")
local hits = tonumber(info:match("keyspace_hits:(%d+)")) or 0
local misses = tonumber(info:match("keyspace_misses:(%d+)")) or 0
local total = hits + misses

if total == 0 then
    return 0
else
    return string.format("%.2f", (hits / total) * 100)
end
